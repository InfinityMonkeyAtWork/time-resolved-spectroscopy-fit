# trsf-dev
development to separate generic 2D time-resolved spectroscopy fitting from the BL931 specific stuff

# getting started
# from the root dir

$ python -m venv venv
.\venv.\Scripts\Activate # [win PowerShell]
source venv/bin/activate # [linux/macOS bash/zsh]
pip install . -e
